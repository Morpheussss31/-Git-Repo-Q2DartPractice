void main() {
  double test = 85.5;
  double testt = 92.0;
  double testtt = 78.5;

  double average;
  String letter;
  String status;

  average = (test + testt + testtt) / 3;

  if (average >= 90) {
    letter = 'A';
  } else if (average >= 85) {
    letter = 'B';
  } else if (average >= 80) {
    letter = 'C';
  } else if (average >= 75) {
    letter = 'D';
  } else if (average >= 70) {
    letter = 'F';
  } else { 
    letter = 'FAIL';
  }
  
  status = average >= 60 ? "PASSED!" : "FAILED!";

  
  print("Test 1: $test");
  print("Test 2: $testt");
  print("Test 3: $testtt");
  print("Average Grade: ${average.toStringAsFixed(2)}");
  print("Status: $status 🎉");
  print("Letter Grade: $letter");
}
